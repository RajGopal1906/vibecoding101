# Step-by-Step Commands for GitHub Push

# FIRST: Install Git from https://git-scm.com/download/win
# Then restart PowerShell and run these commands:

# 1. Configure Git (replace with your email)
git config --global user.name "RajGopal1906"
git config --global user.email "your-email@example.com"

# 2. Navigate to your project
cd "c:\Users\rajgo\OneDrive\Desktop\Vibe Coding 101\website hi"

# 3. Initialize Git repository
git init

# 4. Add all files
git add .

# 5. Create first commit
git commit -m "Initial commit - Pratik Rana blog"

# 6. Rename branch to main
git branch -M main

# 7. Add GitHub repository as remote (create repo first on GitHub!)
git remote add origin https://github.com/RajGopal1906/pratik-rana-blog.git

# 8. Push to GitHub
git push -u origin main
